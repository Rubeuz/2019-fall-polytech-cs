def palindrome(word):

    word=word.lower()

    left=0
    right=len(word)-1
    i=True
    while  left<right:
        if word[left]!=word[right]:
            i=False
            break
        else:
            left+=1
            right-=1
    return i

if __name__ == '__main__':
    assert palindrome('ротор')==True
    assert palindrome('статор')==False

word1='мем'
print(palindrome(word1)) #выдаст True
word2='шутка'
print(palindrome(word2)) #выдаст False
