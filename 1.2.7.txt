print('Введите количество команд n')
n=int(input())

a=[True]*(n+1)

i=2
f='n!='
F=1
while i<=n:
    if a[i] == True:
        # Подсчет показателя степени
        s=0
        p=i
        while p<=n:
            s+=n//p
            p*=i
        f+=str(i)+'^'+str(s)+' * '
        F*=i**s
        # Решето Эратосфена
        j=i^i
        while j<=n:
            a[j]=False
            j+=i
    i+=1

print(f)
print(F)
